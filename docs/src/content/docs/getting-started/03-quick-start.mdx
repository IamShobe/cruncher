import { FileTree } from '@astrojs/starlight/components';
import { Steps } from '@astrojs/starlight/components';

<Steps>

1. Install Cruncher by following the [installation guide](/docs/getting-started/02-installation).
2. Create a configuration file named `cruncher.config.yaml` in your home directory under
   `~/.config/cruncher/`.
   ```bash
     mkdir -p ~/.config/cruncher
   ```
    <FileTree>
    - ~/ user home directory
        - .config/
            - cruncher/
                - **cruncher.config.yaml** cruncher main configuration file
    </FileTree>

</Steps>
